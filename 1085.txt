#include "stdafx.h"
#include <iostream>
#include<iomanip>
#include<cmath>
using namespace std;
int _tmain(int argc, _TCHAR* argv[])
{
	int a, k;
	int j = 0;
	int e[10000];
	int sum = 0;
	while (cin >> a)
	{
		int sum = 0;
		while (a)
		{
			k = a % 10;
			a = a / 10;
			sum = sum + k;
		}
		e[j] = sum;
		j++;
	}
	for (int i = 0; i <= j - 1; i++)
	{
		cout << e[i] << endl;
	}
	
	return 0;
}
