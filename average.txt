#include <iostream>
#include <iomanip>
using namespace std; 
/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main(int argc, char** argv)
{
	int n;
	int t=0;
	int a[100];
	double z[100];
	double sum=0.00;
	double average;
	while(cin>>n)
	{
		double sum=0.00;
		for(int i=0;i<=n-1;i++)
		cin>>a[i];
		for(int k=0;k<=n-1;k++)
		sum=sum+a[k];
		average =sum/double(n);
		z[t]=average;
		t++;
	}
	for(int p=0;p<=t-1;p++)
	{
		cout<<setprecision(3)<<fixed<<z[p]<<endl;
	}
	return 0;
}
