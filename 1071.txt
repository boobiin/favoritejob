#include "stdafx.h"
#include <iostream>
#include<iomanip>
#include<cmath>
using namespace std;
int _tmain(int argc, _TCHAR* argv[])
{
	int a, b, c;
	for (int i = 100; i <= 999; i++)
	{
		a = i % 10;
		b = i / 10 % 10;
		c = i / 10 / 10;
		a = a*a*a;
		b = b*b*b;
		c = c*c*c;
		if (a + b + c == i)
		{
			cout << i << endl;
		}
	}
	
	return 0;
}
